CREATE TABLE DEPT (
    DEPT_ID   INT PRIMARY KEY,
    DEPT_NAME VARCHAR(50),
    DEPT_LOC  VARCHAR(25)
)

CREATE TABLE EMP (
    E_ID    INT PRIMARY KEY,
    E_NAME  VARCHAR(50),
    DEPT_ID INT,
    SALARY  DECIMAL(8, 2),
    FOREIGN KEY ( DEPT_ID )
        REFERENCES DEPT ( DEPT_ID )
)
